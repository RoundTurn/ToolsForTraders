THE TRADE IDEA

This is a mean reversion trade of my own creation, which proved very profitable. ASX bond futures trade 24 hrs. But what do they do during the ASX overnight session? Given Asia is asleep, all they really can do is follow the US bonds around. Mostly.

NOTE this is a not a fully automated trade. Macro must be monitored for divergent CB policy. And please, always have a hard stop out.

Just for fun we are going to ingest 1 minute data, and then resample it to 30.

# We are going to be resampling, and ffilling. So we will need a function to remove the weekends.

def weekend_remover(dts, start_day, start_time, end_day, end_time):
    '''
    datetime.datetime, int, datetime.time(), int, datetime.time() ----> bool
    
    Checks datetime for whether it is within our definition of "the weekend".
    REQUIREMENT datetime. Returns True if timestamp is within the weekend.
    '''
    if dts.weekday() < start_day:
        return False
    elif (dts.weekday() == start_day) & (dts.time() < start_time):
        return False
    elif dts.weekday() > end_day:
        return False
    elif (dts.weekday() == end_day) & (dts.time() > end_time):
        return False
    else:
        return True

class TestStrategy(bt.Strategy):
    params = (
        ('bbandsperiod', 218),
        ('bbdevfactor', 2.0),
        ('bbandsperiod2', 218),
        ('bbdevfactor2', 0.5),
        ('entrytime', datetime.time(14, 0)),
        ('exittime', datetime.time(9, 0)),
        ('maxlong', 7),
        ('maxshort', -7),
        ('stop', 100),
    )
    
    def log(self, txt, dt=None):
        ''' Logging function for this strategy'''
        dt = dt or self.datas[0].datetime.datetime(0)
        # If you want to print out each trade:
#         print('%s, %s' % (dt.isoformat(), txt))

    def __init__(self):
        # Keep a reference to the "close" line in the data[0] dataseries
        self.dataclose = self.datas[0].close
        self.count = len(self)
        # To keep track of pending orders and trade price/commission
        self.order = None
        self.buyprice = None
        self.buycomm = None
        self.sellprice = None
        self.sellcomm = None
        
        # Add indicators
        self.bband = bt.indicators.BBands(self.datas[0], period=self.params.bbandsperiod, 
                                          devfactor=self.params.bbdevfactor)
        self.bband2 = bt.indicators.BBands(self.datas[0], period=self.params.bbandsperiod2, 
                                           devfactor=self.params.bbdevfactor2)

        
        self.wind = pd.to_datetime(('14:30:00', '15:00:00', '15:30:00', 
                                    '16:00:00', '16:30:00'), format='%H:%M:%S')
        self.extimes = pd.to_datetime(('15:12:00', '16:12:00'), format='%H:%M:%S')


    def notify_order(self, order):
        if order.status in [order.Submitted, order.Accepted]:
            # Buy/Sell order submitted/accepted to/by broker - Nothing to do
            return

        # Check if an order has been completed
        # Attention: broker could reject order if not enough cash
        if order.status in [order.Completed]:
            if order.isbuy():
                self.log(
                    'BUY EXECUTED, Price: %.2f, Cost: %.2f, Comm %.2f' %
                    (order.executed.price,
                     order.executed.value,
                     order.executed.comm))

                self.buyprice = order.executed.price
                self.buycomm = order.executed.comm
                self.buytarget = self.buyprice + 10
            else:  # Sell
                self.log('SELL EXECUTED, Price: %.2f, Cost: %.2f, Comm %.2f' %
                         (order.executed.price,
                          order.executed.value,
                          order.executed.comm))
                
                self.sellprice = order.executed.price
                self.sellcomm = order.executed.comm
                self.selltarget = self.sellprice - 10
                
            self.bar_executed = len(self)

        elif order.status in [order.Canceled, order.Margin,]:# order.Rejected]:
            self.log('Order Canceled/Margin/Rejected')
            
        # No pending order
        self.order = None
        
    def notify_trade(self, trade):
        if not trade.isclosed:
            return

        self.log('OPERATION PROFIT, GROSS %.2f, NET %.2f' %
                 (trade.pnl, trade.pnlcomm))
            
    # Timezone converts.
    def utc_to_NSW(self, utc_dt):
        NSW_tz = pytz.timezone('Australia/NSW') 
        NSW_dt = utc_dt.replace(tzinfo=pytz.utc).astimezone(NSW_tz)
        return NSW_tz.normalize(NSW_dt)

    def utc_to_CT(self, utc_dt):
        CT_tz = pytz.timezone('America/Chicago')
        CT_dt = utc_dt.replace(tzinfo=pytz.utc).astimezone(CT_tz)
        return CT_tz.normalize(CT_dt)
    
    
    def next(self):
        # Check if an order is pending ... if yes, we cannot send a 2nd one
        if self.order:
            return

        # Trading rules:
        #BUYS
        if self.dataclose[0] < self.bband.lines.bot and self.position.size < self.params.maxlong:
            if self.utc_to_NSW(self.data.datetime.datetime()).time() in self.wind:
                # Buy (with all possible default parameters).
                self.log('BUY CREATE, %.2f' % (self.dataclose[0]))
                # Keep track of the created order to avoid a 2nd order.
                self.order = self.buy()#size=self.params.maxlong / len(self.extimes))
        elif self.dataclose[0] > self.bband2.lines.top and self.position.size > 0:
            # Close out long.
            self.log('ExitLongTime, %.2f' % self.dataclose[0])
            self.order = self.close()
            
        #SELLS
        if self.dataclose[0] > self.bband.lines.top and self.position.size > self.params.maxshort:
            if self.utc_to_NSW(self.data.datetime.datetime()).time() in self.wind:
                # Sell (with all possible default parameters).
                self.log('SELL CREATE, %.2f' % self.dataclose[0])
                # Keep track of the created order to avoid a 2nd order.
                self.order = self.sell()
        elif self.dataclose[0] < self.bband2.lines.bot and self.position.size < 0:
            # Close out short.
            self.log('ExitShortTime, %.2f' % self.dataclose[0])
            self.order = self.close()
